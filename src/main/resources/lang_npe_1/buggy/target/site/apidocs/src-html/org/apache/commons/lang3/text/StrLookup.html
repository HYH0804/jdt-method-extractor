017package org.apache.commons.lang3.text;
018
019import java.util.Map;
020import java.util.Objects;
021
022import org.apache.commons.lang3.SystemProperties;
023
024/**
025 * Lookup a String key to a String value.
026 * <p>
027 * This class represents the simplest form of a string to string map.
028 * It has a benefit over a map in that it can create the result on
029 * demand based on the key.
030 * </p>
031 * <p>
032 * This class comes complete with various factory methods.
033 * If these do not suffice, you can subclass and implement your own matcher.
034 * </p>
035 * <p>
036 * For example, it would be possible to implement a lookup that used the
037 * key as a primary key, and looked up the value on demand from the database.
038 * </p>
039 *
040 * @param <V> Unused.
041 * @since 2.2
042 * @deprecated As of 3.6, use Apache Commons Text
043 * <a href="https://commons.apache.org/proper/commons-text/javadocs/api-release/org/apache/commons/text/lookup/StringLookupFactory.html">
044 * StringLookupFactory</a> instead
045 */
046@Deprecated
047public abstract class StrLookup<V> {
048
049    /**
050     * Lookup that always returns null.
051     */
052    private static final StrLookup<String> NONE_LOOKUP = new MapStrLookup<>(null);
053
054    /**
055     * Lookup based on system properties.
056     */
057    private static final StrLookup<String> SYSTEM_PROPERTIES_LOOKUP = new SystemPropertiesStrLookup();
058
059    /**
060     * Returns a lookup which always returns null.
061     *
062     * @return a lookup that always returns null, not null
063     */
064    public static StrLookup<?> noneLookup() {
065        return NONE_LOOKUP;
066    }
067
068    /**
069     * Returns a new lookup which uses a copy of the current
070     * {@link System#getProperties() System properties}.
071     * <p>
072     * If a security manager blocked access to system properties, then null will
073     * be returned from every lookup.
074     * </p>
075     * <p>
076     * If a null key is used, this lookup will throw a NullPointerException.
077     * </p>
078     *
079     * @return a lookup using system properties, not null
080     */
081    public static StrLookup<String> systemPropertiesLookup() {
082        return SYSTEM_PROPERTIES_LOOKUP;
083    }
084
085    /**
086     * Returns a lookup which looks up values using a map.
087     * <p>
088     * If the map is null, then null will be returned from every lookup.
089     * The map result object is converted to a string using toString().
090     * </p>
091     *
092     * @param <V> the type of the values supported by the lookup
093     * @param map  the map of keys to values, may be null
094     * @return a lookup using the map, not null
095     */
096    public static <V> StrLookup<V> mapLookup(final Map<String, V> map) {
097        return new MapStrLookup<>(map);
098    }
099
100    /**
101     * Constructor.
102     */
103    protected StrLookup() {
104    }
105
106    /**
107     * Looks up a String key to a String value.
108     * <p>
109     * The internal implementation may use any mechanism to return the value.
110     * The simplest implementation is to use a Map. However, virtually any
111     * implementation is possible.
112     * </p>
113     * <p>
114     * For example, it would be possible to implement a lookup that used the
115     * key as a primary key, and looked up the value on demand from the database
116     * Or, a numeric based implementation could be created that treats the key
117     * as an integer, increments the value and return the result as a string -
118     * converting 1 to 2, 15 to 16 etc.
119     * </p>
120     * <p>
121     * The {@link #lookup(String)} method always returns a String, regardless of
122     * the underlying data, by converting it as necessary. For example:
123     * </p>
124     * <pre>
125     * Map&lt;String, Object&gt; map = new HashMap&lt;String, Object&gt;();
126     * map.put("number", Integer.valueOf(2));
127     * assertEquals("2", StrLookup.mapLookup(map).lookup("number"));
128     * </pre>
129     * @param key  the key to be looked up, may be null
130     * @return the matching value, null if no match
131     */
132    public abstract String lookup(String key);
133
134    /**
135     * Lookup implementation that uses a Map.
136     *
137     * @param <V> the type of mapped values.
138     */
139    static class MapStrLookup<V> extends StrLookup<V> {
140
141        /** Map keys are variable names and value. */
142        private final Map<String, V> map;
143
144        /**
145         * Creates a new instance backed by a Map.
146         *
147         * @param map  the map of keys to values, may be null
148         */
149        MapStrLookup(final Map<String, V> map) {
150            this.map = map;
151        }
152
153        /**
154         * Looks up a String key to a String value using the map.
155         * <p>
156         * If the map is null, then null is returned.
157         * The map result object is converted to a string using toString().
158         * </p>
159         *
160         * @param key  the key to be looked up, may be null
161         * @return the matching value, null if no match
162         */
163        @Override
164        public String lookup(final String key) {
165            if (map == null) {
166                return null;
167            }
168            return Objects.toString(map.get(key), null);
169        }
170    }
171
172    /**
173     * Lookup implementation based on system properties.
174     */
175    private static class SystemPropertiesStrLookup extends StrLookup<String> {
176        /**
177         * {@inheritDoc} This implementation directly accesses system properties.
178         */
179        @Override
180        public String lookup(final String key) {
181            return SystemProperties.getProperty(key);
182        }
183    }
184}
