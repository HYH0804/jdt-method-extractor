031/**
032 * Duration formatting utilities and constants. The following table describes the tokens
033 * used in the pattern language for formatting.
034 * <table border="1">
035 *  <caption>Pattern Tokens</caption>
036 *  <tr><th>character</th><th>duration element</th></tr>
037 *  <tr><td>y</td><td>years</td></tr>
038 *  <tr><td>M</td><td>months</td></tr>
039 *  <tr><td>d</td><td>days</td></tr>
040 *  <tr><td>H</td><td>hours</td></tr>
041 *  <tr><td>m</td><td>minutes</td></tr>
042 *  <tr><td>s</td><td>seconds</td></tr>
043 *  <tr><td>S</td><td>milliseconds</td></tr>
044 *  <tr><td>'text'</td><td>arbitrary text content</td></tr>
045 * </table>
046 *
047 * <b>Note: It's not currently possible to include a single-quote in a format.</b>
048 * <br>
049 * Token values are printed using decimal digits.
050 * A token character can be repeated to ensure that the field occupies a certain minimum
051 * size. Values will be left-padded with 0 unless padding is disabled in the method invocation.
052 * <br>
053 * Tokens can be marked as optional by surrounding them with brackets [ ]. These tokens will
054 * only be printed if the token value is non-zero. Literals within optional blocks will only be
055 * printed if the preceding non-literal token is non-zero. Leading optional literals will only
056 * be printed if the following non-literal is non-zero.
057 * Multiple optional blocks can be used to group literals with the desired token.
058 * <p>
059 * Notes on Optional Tokens:<br>
060 * <b>Multiple optional tokens without literals can result in impossible to understand output.</b><br>
061 * <b>Patterns where all tokens are optional can produce empty strings.</b><br>
062 * (See examples below)
063 * </p>
064 * <br>
065 * <table border="1">
066 * <caption>Example Output</caption>
067 * <tr><th>pattern</th><th>Duration.ofDays(1)</th><th>Duration.ofHours(1)</th><th>Duration.ofMinutes(1)</th><th>Duration.ZERO</th></tr>
068 * <tr><td>d'd'H'h'm'm's's'</td><td>1d0h0m0s</td><td>0d1h0m0s</td><td>0d0h1m0s</td><td>0d0h0m0s</td></tr>
069 * <tr><td>d'd'[H'h'm'm']s's'</td><td>1d0s</td><td>0d1h0s</td><td>0d1m0s</td><td>0d0s</td></tr>
070 * <tr><td>[d'd'H'h'm'm']s's'</td><td>1d0s</td><td>1h0s</td><td>1m0s</td><td>0s</td></tr>
071 * <tr><td>[d'd'H'h'm'm's's']</td><td>1d</td><td>1h</td><td>1m</td><td></td></tr>
072 * <tr><td>['{'d'}']HH':'mm</td><td>{1}00:00</td><td>01:00</td><td>00:01</td><td>00:00</td></tr>
073 * <tr><td>['{'dd'}']['&lt;'HH'&gt;']['('mm')']</td><td>{01}</td><td>&lt;01&gt;</td><td>(00)</td><td></td></tr>
074 * <tr><td>[dHms]</td><td>1</td><td>1</td><td>1</td><td></td></tr>
075 * </table>
076 * <b>Note: Optional blocks cannot be nested.</b>
077 *
078 * @since 2.1
079 */

144    /**
145     * Formats the time gap as a string, using the specified format.
146     * Padding the left-hand side of numbers with zeroes is optional.
147     *
148     * <p>This method formats durations using the days and lower fields of the
149     * format pattern. Months and larger are not used.</p>
150     *
151     * @param durationMillis  the duration to format
152     * @param format  the way in which to format the duration, not null
153     * @param padWithZeros  whether to pad the left-hand side of numbers with 0's
154     * @return the formatted duration, not null
155     * @throws IllegalArgumentException if durationMillis is negative
156     */
157    public static String formatDuration(final long durationMillis, final String format, final boolean padWithZeros) {
158        Validate.inclusiveBetween(0, Long.MAX_VALUE, durationMillis, "durationMillis must not be negative");
159
160        final Token[] tokens = lexx(format);
161
162        long days = 0;
163        long hours = 0;
164        long minutes = 0;
165        long seconds = 0;
166        long milliseconds = durationMillis;
167
168        if (Token.containsTokenWithValue(tokens, d)) {
169            days = milliseconds / DateUtils.MILLIS_PER_DAY;
170            milliseconds = milliseconds - (days * DateUtils.MILLIS_PER_DAY);
171        }
172        if (Token.containsTokenWithValue(tokens, H)) {
173            hours = milliseconds / DateUtils.MILLIS_PER_HOUR;
174            milliseconds = milliseconds - (hours * DateUtils.MILLIS_PER_HOUR);
175        }
176        if (Token.containsTokenWithValue(tokens, m)) {
177            minutes = milliseconds / DateUtils.MILLIS_PER_MINUTE;
178            milliseconds = milliseconds - (minutes * DateUtils.MILLIS_PER_MINUTE);
179        }
180        if (Token.containsTokenWithValue(tokens, s)) {
181            seconds = milliseconds / DateUtils.MILLIS_PER_SECOND;
182            milliseconds = milliseconds - (seconds * DateUtils.MILLIS_PER_SECOND);
183        }
184
185        return format(tokens, 0, 0, days, hours, minutes, seconds, milliseconds, padWithZeros);
186    }

188    /**
189     * Formats an elapsed time into a pluralization correct string.
190     *
191     * <p>This method formats durations using the days and lower fields of the
192     * format pattern. Months and larger are not used.</p>
193     *
194     * @param durationMillis  the elapsed time to report in milliseconds
195     * @param suppressLeadingZeroElements  suppresses leading 0 elements
196     * @param suppressTrailingZeroElements  suppresses trailing 0 elements
197     * @return the formatted text in days/hours/minutes/seconds, not null
198     * @throws IllegalArgumentException if durationMillis is negative
199     */
200    public static String formatDurationWords(
201        final long durationMillis,
202        final boolean suppressLeadingZeroElements,
203        final boolean suppressTrailingZeroElements) {
204
205        // This method is generally replaceable by the format method, but
206        // there are a series of tweaks and special cases that require
207        // trickery to replicate.
208        String duration = formatDuration(durationMillis, "d' days 'H' hours 'm' minutes 's' seconds'");
209        if (suppressLeadingZeroElements) {
210            // this is a temporary marker on the front. Like ^ in regexp.
211            duration = " " + duration;
212            String tmp = StringUtils.replaceOnce(duration, " 0 days", StringUtils.EMPTY);
213            if (tmp.length() != duration.length()) {
214                duration = tmp;
215                tmp = StringUtils.replaceOnce(duration, " 0 hours", StringUtils.EMPTY);
216                if (tmp.length() != duration.length()) {
217                    duration = tmp;
218                    tmp = StringUtils.replaceOnce(duration, " 0 minutes", StringUtils.EMPTY);
219                    duration = tmp;
220                }
221            }
222            if (!duration.isEmpty()) {
223                // strip the space off again
224                duration = duration.substring(1);
225            }
226        }
227        if (suppressTrailingZeroElements) {
228            String tmp = StringUtils.replaceOnce(duration, " 0 seconds", StringUtils.EMPTY);
229            if (tmp.length() != duration.length()) {
230                duration = tmp;
231                tmp = StringUtils.replaceOnce(duration, " 0 minutes", StringUtils.EMPTY);
232                if (tmp.length() != duration.length()) {
233                    duration = tmp;
234                    tmp = StringUtils.replaceOnce(duration, " 0 hours", StringUtils.EMPTY);
235                    if (tmp.length() != duration.length()) {
236                        duration = StringUtils.replaceOnce(tmp, " 0 days", StringUtils.EMPTY);
237                    }
238                }
239            }
240        }
241        // handle plurals
242        duration = " " + duration;
243        duration = StringUtils.replaceOnce(duration, " 1 seconds", " 1 second");
244        duration = StringUtils.replaceOnce(duration, " 1 minutes", " 1 minute");
245        duration = StringUtils.replaceOnce(duration, " 1 hours", " 1 hour");
246        duration = StringUtils.replaceOnce(duration, " 1 days", " 1 day");
247        return duration.trim();
248    }

278    /**
279     * <p>Formats the time gap as a string, using the specified format.
280     * Padding the left-hand side of numbers with zeroes is optional and
281     * the time zone may be specified.
282     *
283     * <p>When calculating the difference between months/days, it chooses to
284     * calculate months first. So when working out the number of months and
285     * days between January 15th and March 10th, it choose 1 month and
286     * 23 days gained by choosing January-&gt;February = 1 month and then
287     * calculating days forwards, and not the 1 month and 26 days gained by
288     * choosing March -&gt; February = 1 month and then calculating days
289     * backwards.</p>
290     *
291     * <p>For more control, the <a href="https://www.joda.org/joda-time/">Joda-Time</a>
292     * library is recommended.</p>
293     *
294     * @param startMillis  the start of the duration
295     * @param endMillis  the end of the duration
296     * @param format  the way in which to format the duration, not null
297     * @param padWithZeros  whether to pad the left-hand side of numbers with 0's
298     * @param timezone  the millis are defined in
299     * @return the formatted duration, not null
300     * @throws IllegalArgumentException if startMillis is greater than endMillis
301     */
302    public static String formatPeriod(final long startMillis, final long endMillis, final String format, final boolean padWithZeros,
303            final TimeZone timezone) {
304        Validate.isTrue(startMillis <= endMillis, "startMillis must not be greater than endMillis");
305
306
307        // Used to optimise for differences under 28 days and
308        // called formatDuration(millis, format); however this did not work
309        // over leap years.
310        // TODO: Compare performance to see if anything was lost by
311        // losing this optimisation.
312
313        final Token[] tokens = lexx(format);
314
315        // time zones get funky around 0, so normalizing everything to GMT
316        // stops the hours being off
317        final Calendar start = Calendar.getInstance(timezone);
318        start.setTime(new Date(startMillis));
319        final Calendar end = Calendar.getInstance(timezone);
320        end.setTime(new Date(endMillis));
321
322        // initial estimates
323        int milliseconds = end.get(Calendar.MILLISECOND) - start.get(Calendar.MILLISECOND);
324        int seconds = end.get(Calendar.SECOND) - start.get(Calendar.SECOND);
325        int minutes = end.get(Calendar.MINUTE) - start.get(Calendar.MINUTE);
326        int hours = end.get(Calendar.HOUR_OF_DAY) - start.get(Calendar.HOUR_OF_DAY);
327        int days = end.get(Calendar.DAY_OF_MONTH) - start.get(Calendar.DAY_OF_MONTH);
328        int months = end.get(Calendar.MONTH) - start.get(Calendar.MONTH);
329        int years = end.get(Calendar.YEAR) - start.get(Calendar.YEAR);
330
331        // each initial estimate is adjusted in case it is under 0
332        while (milliseconds < 0) {
333            milliseconds += 1000;
334            seconds -= 1;
335        }
336        while (seconds < 0) {
337            seconds += 60;
338            minutes -= 1;
339        }
340        while (minutes < 0) {
341            minutes += 60;
342            hours -= 1;
343        }
344        while (hours < 0) {
345            hours += 24;
346            days -= 1;
347        }
348
349        if (Token.containsTokenWithValue(tokens, M)) {
350            while (days < 0) {
351                days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
352                months -= 1;
353                start.add(Calendar.MONTH, 1);
354            }
355
356            while (months < 0) {
357                months += 12;
358                years -= 1;
359            }
360
361            if (!Token.containsTokenWithValue(tokens, y) && years != 0) {
362                while (years != 0) {
363                    months += 12 * years;
364                    years = 0;
365                }
366            }
367        } else {
368            // there are no M's in the format string
369
370            if (!Token.containsTokenWithValue(tokens, y)) {
371                int target = end.get(Calendar.YEAR);
372                if (months < 0) {
373                    // target is end-year -1
374                    target -= 1;
375                }
376
377                while (start.get(Calendar.YEAR) != target) {
378                    days += start.getActualMaximum(Calendar.DAY_OF_YEAR) - start.get(Calendar.DAY_OF_YEAR);
379
380                    // Not sure I grok why this is needed, but the brutal tests show it is
381                    if (start instanceof GregorianCalendar &&
382                            start.get(Calendar.MONTH) == Calendar.FEBRUARY &&
383                            start.get(Calendar.DAY_OF_MONTH) == 29) {
384                        days += 1;
385                    }
386
387                    start.add(Calendar.YEAR, 1);
388
389                    days += start.get(Calendar.DAY_OF_YEAR);
390                }
391
392                years = 0;
393            }
394
395            while (start.get(Calendar.MONTH) != end.get(Calendar.MONTH)) {
396                days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
397                start.add(Calendar.MONTH, 1);
398            }
399
400            months = 0;
401
402            while (days < 0) {
403                days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
404                months -= 1;
405                start.add(Calendar.MONTH, 1);
406            }
407
408        }
409
410        // The rest of this code adds in values that
411        // aren't requested. This allows the user to ask for the
412        // number of months and get the real count and not just 0->11.
413
414        if (!Token.containsTokenWithValue(tokens, d)) {
415            hours += 24 * days;
416            days = 0;
417        }
418        if (!Token.containsTokenWithValue(tokens, H)) {
419            minutes += 60 * hours;
420            hours = 0;
421        }
422        if (!Token.containsTokenWithValue(tokens, m)) {
423            seconds += 60 * minutes;
424            minutes = 0;
425        }
426        if (!Token.containsTokenWithValue(tokens, s)) {
427            milliseconds += 1000 * seconds;
428            seconds = 0;
429        }
430
431        return format(tokens, years, months, days, hours, minutes, seconds, milliseconds, padWithZeros);
432    }

434    /**
435     * The internal method to do the formatting.
436     *
437     * @param tokens  the tokens
438     * @param years  the number of years
439     * @param months  the number of months
440     * @param days  the number of days
441     * @param hours  the number of hours
442     * @param minutes  the number of minutes
443     * @param seconds  the number of seconds
444     * @param milliseconds  the number of millis
445     * @param padWithZeros  whether to pad
446     * @return the formatted string
447     */
448    static String format(final Token[] tokens, final long years, final long months, final long days, final long hours, final long minutes,
449            final long seconds,
450            final long milliseconds, final boolean padWithZeros) {
451        final StringBuilder buffer = new StringBuilder();
452        boolean lastOutputSeconds = false;
453        boolean lastOutputZero = false;
454        int optionalStart = -1;
455        boolean firstOptionalNonLiteral = false;
456        int optionalIndex = -1;
457        boolean inOptional = false;
458        for (final Token token : tokens) {
459            final Object value = token.getValue();
460            final boolean isLiteral = value instanceof StringBuilder;
461            final int count = token.getCount();
462            if (optionalIndex != token.optionalIndex) {
463              optionalIndex = token.optionalIndex;
464              if (optionalIndex > -1) {
465                //entering new optional block
466                optionalStart = buffer.length();
467                lastOutputZero = false;
468                inOptional = true;
469                firstOptionalNonLiteral = false;
470              } else {
471                //leaving optional block
472                inOptional = false;
473              }
474            }
475            if (isLiteral) {
476               if (!inOptional || !lastOutputZero) {
477                     buffer.append(value.toString());
478               }
479            } else if (value.equals(y)) {
480                lastOutputSeconds = false;
481                lastOutputZero = years == 0;
482                if (!inOptional || !lastOutputZero) {
483                    buffer.append(paddedValue(years, padWithZeros, count));
484                }
485            } else if (value.equals(M)) {
486                lastOutputSeconds = false;
487                lastOutputZero = months == 0;
488                if (!inOptional || !lastOutputZero) {
489                    buffer.append(paddedValue(months, padWithZeros, count));
490                }
491            } else if (value.equals(d)) {
492                lastOutputSeconds = false;
493                lastOutputZero = days == 0;
494                if (!inOptional || !lastOutputZero) {
495                    buffer.append(paddedValue(days, padWithZeros, count));
496                }
497            } else if (value.equals(H)) {
498                lastOutputSeconds = false;
499                lastOutputZero = hours == 0;
500                if (!inOptional || !lastOutputZero) {
501                    buffer.append(paddedValue(hours, padWithZeros, count));
502                }
503            } else if (value.equals(m)) {
504                lastOutputSeconds = false;
505                lastOutputZero = minutes == 0;
506                if (!inOptional || !lastOutputZero) {
507                    buffer.append(paddedValue(minutes, padWithZeros, count));
508                }
509            } else if (value.equals(s)) {
510                lastOutputSeconds = true;
511                lastOutputZero = seconds == 0;
512                if (!inOptional || !lastOutputZero) {
513                    buffer.append(paddedValue(seconds, padWithZeros, count));
514                }
515            } else if (value.equals(S)) {
516                lastOutputZero = milliseconds == 0;
517                if (!inOptional || !lastOutputZero) {
518                    if (lastOutputSeconds) {
519                        // ensure at least 3 digits are displayed even if padding is not selected
520                        final int width = padWithZeros ? Math.max(3, count) : 3;
521                        buffer.append(paddedValue(milliseconds, true, width));
522                    } else {
523                        buffer.append(paddedValue(milliseconds, padWithZeros, count));
524                    }
525                }
526                lastOutputSeconds = false;
527            }
528            //as soon as we hit first nonliteral in optional, check for literal prefix
529            if (inOptional && !isLiteral && !firstOptionalNonLiteral){
530                firstOptionalNonLiteral = true;
531                if (lastOutputZero) {
532                    buffer.delete(optionalStart, buffer.length());
533                }
534            }
535        }
536        return buffer.toString();
537    }

539    /**
540     * Converts a {@code long} to a {@link String} with optional
541     * zero padding.
542     *
543     * @param value the value to convert
544     * @param padWithZeros whether to pad with zeroes
545     * @param count the size to pad to (ignored if {@code padWithZeros} is false)
546     * @return the string result
547     */
548    private static String paddedValue(final long value, final boolean padWithZeros, final int count) {
549        final String longString = Long.toString(value);
550        return padWithZeros ? StringUtils.leftPad(longString, count, '0') : longString;
551    }

561    /**
562     * Parses a classic date format string into Tokens
563     *
564     * @param format  the format to parse, not null
565     * @return array of Token[]
566     */
567    static Token[] lexx(final String format) {
568        final ArrayList<Token> list = new ArrayList<>(format.length());
569
570        boolean inLiteral = false;
571        // Although the buffer is stored in a Token, the Tokens are only
572        // used internally, so cannot be accessed by other threads
573        StringBuilder buffer = null;
574        Token previous = null;
575        boolean inOptional = false;
576        int optionalIndex = -1;
577        for (int i = 0; i < format.length(); i++) {
578            final char ch = format.charAt(i);
579            if (inLiteral && ch != '\'') {
580                buffer.append(ch); // buffer can't be null if inLiteral is true
581                continue;
582            }
583            String value = null;
584            switch (ch) {
585            // TODO: Need to handle escaping of '
586            case '[':
587                if (inOptional) {
588                  throw new IllegalArgumentException("Nested optional block at index: "+i);
589                }
590                optionalIndex++;
591                inOptional = true;
592                break;
593            case ']':
594                if (!inOptional) {
595                  throw new IllegalArgumentException("Attempting to close unopened optional block at index: "+i);
596                }
597                inOptional = false;
598                break;
599            case '\'':
600                if (inLiteral) {
601                    buffer = null;
602                    inLiteral = false;
603                } else {
604                    buffer = new StringBuilder();
605                    list.add(new Token(buffer, inOptional, optionalIndex));
606                    inLiteral = true;
607                }
608                break;
609            case 'y':
610                value = y;
611                break;
612            case 'M':
613                value = M;
614                break;
615            case 'd':
616                value = d;
617                break;
618            case 'H':
619                value = H;
620                break;
621            case 'm':
622                value = m;
623                break;
624            case 's':
625                value = s;
626                break;
627            case 'S':
628                value = S;
629                break;
630            default:
631                if (buffer == null) {
632                    buffer = new StringBuilder();
633                    list.add(new Token(buffer, inOptional, optionalIndex));
634                }
635                buffer.append(ch);
636            }
637
638            if (value != null) {
639                if (previous != null && previous.getValue().equals(value)) {
640                    previous.increment();
641                } else {
642                    final Token token = new Token(value, inOptional, optionalIndex);
643                    list.add(token);
644                    previous = token;
645                }
646                buffer = null;
647            }
648        }
649        if (inLiteral) { // i.e. we have not found the end of the literal
650            throw new IllegalArgumentException("Unmatched quote in format: " + format);
651        }
652        if (inOptional) { // i.e. we have not found the end of the literal
653            throw new IllegalArgumentException("Unmatched optional in format: " + format);
654        }
655        return list.toArray(Token.EMPTY_ARRAY);
656    }
657
658    /**
659     * Element that is parsed from the format pattern.
660     */
661    static class Token {
662
663        /** Empty array. */
664        private static final Token[] EMPTY_ARRAY = {};
665
666        /**
667         * Helper method to determine if a set of tokens contain a value
668         *
669         * @param tokens set to look in
670         * @param value to look for
671         * @return boolean {@code true} if contained
672         */
673        static boolean containsTokenWithValue(final Token[] tokens, final Object value) {
674            return Stream.of(tokens).anyMatch(token -> token.getValue() == value);
675        }
676
677        private final Object value;
678        private int count;
679        private int optionalIndex = -1;
680
681        /**
682         * Wraps a token around a value. A value would be something like a 'Y'.
683         *
684         * @param value value to wrap, non-null.
685         * @param optional whether the token is optional
686         * @param optionalIndex the index of the optional token within the pattern
687         */
688        Token(final Object value, final boolean optional, final int optionalIndex) {
689            this.value = Objects.requireNonNull(value, "value");
690            this.count = 1;
691            if (optional) {
692                this.optionalIndex = optionalIndex;
693            }
694        }
695
696        /**
697         * Adds another one of the value
698         */
699        void increment() {
700            count++;
701        }
702
703        /**
704         * Gets the current number of values represented
705         *
706         * @return int number of values represented
707         */
708        int getCount() {
709            return count;
710        }
711
712        /**
713         * Gets the particular value this token represents.
714         *
715         * @return Object value, non-null.
716         */
717        Object getValue() {
718            return value;
719        }
720
721        /**
722         * Supports equality of this Token to another Token.
723         *
724         * @param obj2 Object to consider equality of
725         * @return boolean {@code true} if equal
726         */
727        @Override
728        public boolean equals(final Object obj2) {
729            if (obj2 instanceof Token) {
730                final Token tok2 = (Token) obj2;
731                if (this.value.getClass() != tok2.value.getClass()) {
732                    return false;
733                }
734                if (this.count != tok2.count) {
735                    return false;
736                }
737                if (this.value instanceof StringBuilder) {
738                    return this.value.toString().equals(tok2.value.toString());
739                }
740                if (this.value instanceof Number) {
741                    return this.value.equals(tok2.value);
742                }
743                return this.value == tok2.value;
744            }
745            return false;
746        }
747
748        /**
749         * Returns a hash code for the token equal to the
750         * hash code for the token's value. Thus 'TT' and 'TTTT'
751         * will have the same hash code.
752         *
753         * @return The hash code for the token
754         */
755        @Override
756        public int hashCode() {
757            return this.value.hashCode();
758        }
759
760        /**
761         * Represents this token as a String.
762         *
763         * @return String representation of the token
764         */
765        @Override
766        public String toString() {
767            return StringUtils.repeat(this.value.toString(), this.count);
768        }
769    }
