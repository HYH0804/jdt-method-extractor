2096    /**
2097     * Decides if the operating system version matches.
2098     * <p>
2099     * This method is package private instead of private to support unit test invocation.
2100     * </p>
2101     *
2102     * @param osVersion the actual OS version
2103     * @param osVersionPrefix the prefix for the expected OS version
2104     * @return true if matches, or false if not or can't determine
2105     */
2106    static boolean isOSVersionMatch(final String osVersion, final String osVersionPrefix) {
2107        if (StringUtils.isEmpty(osVersion)) {
2108            return false;
2109        }
2110        // Compare parts of the version string instead of using String.startsWith(String) because otherwise
2111        // osVersionPrefix 10.1 would also match osVersion 10.10
2112        final String[] versionPrefixParts = osVersionPrefix.split("\\.");
2113        final String[] versionParts = osVersion.split("\\.");
2114        for (int i = 0; i < Math.min(versionPrefixParts.length, versionParts.length); i++) {
2115            if (!versionPrefixParts[i].equals(versionParts[i])) {
2116                return false;
2117            }
2118        }
2119        return true;
2120    }
