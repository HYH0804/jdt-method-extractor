017package org.apache.commons.lang3.stream;
018
019import java.util.stream.IntStream;
020
021/**
022 * Factory for {@link IntStream}.
023 * <p>
024 * <small> Only a factory for now but could hold other functionality.</small>
025 * </p>
026 *
027 * @since 3.13.0
028 */
029public class IntStreams {
030
031    /**
032     * Shorthand for {@code IntStream.range(0, i)}.
033     *
034     * @param endExclusive the exclusive upper bound.
035     * @return a sequential {@link IntStream} for the range of {@code int} elements.
036     */
037    public static IntStream range(final int endExclusive) {
038        return IntStream.range(0, endExclusive);
039    }
040
041    /**
042     * Shorthand for {@code IntStream.rangeClosed(0, i)}.
043     *
044     * @param endInclusive the inclusive upper bound.
045     * @return a sequential {@link IntStream} for the range of {@code int} elements.
046     */
047    public static IntStream rangeClosed(final int endInclusive) {
048        return IntStream.rangeClosed(0, endInclusive);
049    }
050
051}
