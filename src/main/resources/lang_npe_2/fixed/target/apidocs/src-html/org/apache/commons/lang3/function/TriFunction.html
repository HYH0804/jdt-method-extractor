017package org.apache.commons.lang3.function;
018
019import java.util.Objects;
020import java.util.function.Function;
021
022/**
023 * Represents a function that accepts three arguments and produces a result. This is the three-arity specialization of
024 * {@link Function}.
025 *
026 * <p>
027 * This is a <a href="package-summary.html">functional interface</a> whose functional method is
028 * {@link #apply(Object, Object, Object)}.
029 * </p>
030 *
031 * @param <T> the type of the first argument to the function
032 * @param <U> the type of the second argument to the function
033 * @param <V> the type of the third argument to the function
034 * @param <R> the type of the result of the function
035 *
036 * @see Function
037 * @since 3.12.0
038 */
039@FunctionalInterface
040public interface TriFunction<T, U, V, R> {
041
042    /**
043     * Applies this function to the given arguments.
044     *
045     * @param t the first function argument
046     * @param u the second function argument
047     * @param v the third function argument
048     * @return the function result
049     */
050    R apply(T t, U u, V v);
051
052    /**
053     * Returns a composed function that first applies this function to its input, and then applies the {@code after}
054     * function to the result. If evaluation of either function throws an exception, it is relayed to the caller of the
055     * composed function.
056     *
057     * @param <W> the type of output of the {@code after} function, and of the composed function
058     * @param after the function to apply after this function is applied
059     * @return a composed function that first applies this function and then applies the {@code after} function
060     * @throws NullPointerException if after is null
061     */
062    default <W> TriFunction<T, U, V, W> andThen(final Function<? super R, ? extends W> after) {
063        Objects.requireNonNull(after);
064        return (final T t, final U u, final V v) -> after.apply(apply(t, u, v));
065    }
066}
