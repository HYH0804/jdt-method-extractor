017package org.apache.commons.lang3.concurrent;
018
019/**
020 * An exception class used for reporting runtime error conditions related to
021 * accessing data of background tasks.
022 *
023 * <p>
024 * This class is an analogue of the {@link ConcurrentException} exception class.
025 * However, it is a runtime exception and thus does not need explicit catch
026 * clauses. Some methods of {@link ConcurrentUtils} throw {@code
027 * ConcurrentRuntimeException} exceptions rather than
028 * {@link ConcurrentException} exceptions. They can be used by client code that
029 * does not want to be bothered with checked exceptions.
030 * </p>
031 *
032 * @since 3.0
033 */
034public class ConcurrentRuntimeException extends RuntimeException {
035    /**
036     * The serial version UID.
037     */
038    private static final long serialVersionUID = -6582182735562919670L;
039
040    /**
041     * Creates a new, uninitialized instance of {@code
042     * ConcurrentRuntimeException}.
043     */
044    protected ConcurrentRuntimeException() {
045    }
046
047    /**
048     * Creates a new instance of {@link ConcurrentRuntimeException} and
049     * initializes it with the given cause.
050     *
051     * @param cause the cause of this exception
052     * @throws IllegalArgumentException if the cause is not a checked exception
053     */
054    public ConcurrentRuntimeException(final Throwable cause) {
055        super(ConcurrentUtils.checkedException(cause));
056    }
057
058    /**
059     * Creates a new instance of {@link ConcurrentRuntimeException} and
060     * initializes it with the given message and cause.
061     *
062     * @param msg the error message
063     * @param cause the cause of this exception
064     * @throws IllegalArgumentException if the cause is not a checked exception
065     */
066    public ConcurrentRuntimeException(final String msg, final Throwable cause) {
067        super(msg, ConcurrentUtils.checkedException(cause));
068    }
069}
